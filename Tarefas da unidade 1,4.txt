Exercício 3: Média Aritmética de um Array

function calcularMedia(numeros) {
  // Verifica se o array está vazio para evitar divisão por zero
  if (numeros.length === 0) {
    return 0;
  }
  
  // Soma todos os números do array usando o método reduce
  const soma = numeros.reduce((acumulador, numeroAtual) => acumulador + numeroAtual, 0);
  
  // Divide a soma pela quantidade de elementos
  return soma / numeros.length;
}

// Exemplos de uso

console.log(`A média de outras notas é: ${calcularMedia(outrasNotas)}`); // Saída: 8

Exercício 5: Número Primo

function ehPrimo(numero) {
  // Números menores ou iguais a 1 não são primos
  if (numero <= 1) {
    return false;
  }

  // Verifica divisores de 2 até a raiz quadrada do número
  for (let i = 2; i <= Math.sqrt(numero); i++) {
    if (numero % i === 0) {
      // Se encontrar um divisor, não é primo
      return false;
    }
  }

  // Se não encontrar divisores, é primo
  return true;
}

// Exemplos de uso
console.log("O número 7 é primo?", ehPrimo(7));     // Saída: true
console.log("O número 15 é primo?", ehPrimo(15));   // Saída: false
console.log("O número 29 é primo?", ehPrimo(29));   // Saída: true
console.log("O número 1 é primo?", ehPrimo(1));     // Saída: false

function ehPrimo(numero) {
  // Números menores ou iguais a 1 não são primos
  if (numero <= 1) {
    return false;
  }

  // Verifica divisores de 2 até a raiz quadrada do número
  for (let i = 2; i <= Math.sqrt(numero); i++) {
    if (numero % i === 0) {
      // Se encontrar um divisor, não é primo
      return false;
    }
  }

  // Se não encontrar divisores, é primo
  return true;
}

// Exemplos de uso
console.log("O número 7 é primo?", ehPrimo(7));     // Saída: true
console.log("O número 15 é primo?", ehPrimo(15));   // Saída: false
console.log("O número 29 é primo?", ehPrimo(29));   // Saída: true
console.log("O número 1 é primo?", ehPrimo(1));     // Saída: false
Explicação: Um número primo é divisível apenas por 1 e por ele mesmo. A função testa se o número tem algum divisor entre 2 e sua raiz quadrada. Se encontrar algum, o número não é primo.

Exercício 10: Fatorial de um Número

function calcularFatorial(numero) {
  // Casos base: fatorial de 0 e 1 é 1. Números negativos não têm fatorial definido.
  if (numero < 0) {
    return "Não é possível calcular o fatorial de um número negativo.";
  }
  if (numero === 0 || numero === 1) {
    return 1;
  }

  // Lógica para calcular o fatorial
  let resultado = 1;
  for (let i = 2; i <= numero; i++) {
    resultado *= i;
  }
  
  return resultado;
}

// Exemplos de uso
console.log("O fatorial de 5 é:", calcularFatorial(5));   // Saída: 120 (5*4*3*2*1)
console.log("O fatorial de 0 é:", calcularFatorial(0));   // Saída: 1
console.log("O fatorial de 7 é:", calcularFatorial(7));   // Saída: 5040
console.log("O fatorial de -3 é:", calcularFatorial(-3)); // Saída: "Não é possível calcular o fatorial de um número negativo."
